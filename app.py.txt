import streamlit as st
import pandas as pd

st.title("Flu Prediction System - Cauayan City")
st.subheader("Upload flu and weather CSV files to see predictions")

# Load sample data
flu_file = st.file_uploader("Upload Flu Cases CSV", type=["csv"])
weather_file = st.file_uploader("Upload Weather CSV", type=["csv"])

if flu_file and weather_file:
    flu_data = pd.read_csv(flu_file)
    weather_data = pd.read_csv(weather_file)

    st.write("### Flu Data")
    st.dataframe(flu_data)

    st.write("### Weather Data")
    st.dataframe(weather_data)

    # Dummy prediction example: just add 1 to Dengue_Cases
    merged = pd.merge(flu_data, weather_data, on=["Date", "Barangay"])
    merged['Predicted_Cases'] = merged['Dengue_Cases'] + 1
    st.write("### Predicted Cases")
    st.dataframe(merged[['Date', 'Barangay', 'Predicted_Cases']])
else:
    st.info("Please upload both CSV files to see predictions. Sample CSVs included below.")
